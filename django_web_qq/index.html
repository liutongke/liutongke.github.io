    def get_auth_url(self):
        """è·å–æˆæƒé¡µé¢çš„ç½‘å€"""

    def get_access_token(self, code):
        """æ ¹æ®codeè·å–access_token"""
        url = "https://graph.qq.com/oauth2.0/token"

        querystring = {
            "grant_type": "authorization_code",
            "client_id": self.client_id,
            "client_secret": self.client_key,
            "code": code,
            "redirect_uri": self.redirect_uri
        }

        response = requests.request("GET", url, params=querystring)
        return response.text.split('=', 3)[1].split('&', 1)[0]

    def get_openid(self, access_token):  # è·å–ç”¨æˆ·çš„openid

        querystring = {"access_token": access_token}

        response = requests.request("GET", url, params=querystring)
        call_back = str(response.text)[9:-3]
        return json.loads(call_back)['openid']

    def get_user_info(self, access_token, qq_openid):  # è·å–ç”¨æˆ·çš„è¯¦ç»†ä¿¡æ¯

            "openid": qq_openid

        response = requests.request("GET", url, params=querystring)

        return json.loads(response.text)
